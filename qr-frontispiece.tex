\chapter*{Nomenclature}

\begin{tabulary}{\textwidth}{lL}
\code{\\foo} &
\TeX{} control sequence. \\

\env{foo} &
\LaTeX{} environment \code{foo}. \\

\sty{foo} &
\PGFPlots{} style with key \code{foo}. \\

\code{foo} &
Terminal symbol, literal text. \\

\metavar{foo} &
Non-terminal symbol, metasyntactic variable. \\

\hbox{\metavar{foo}\metadef\metavar{bar}} &
Production rule; \metavar{foo} can be replaced by \metavar{bar},
\metavar{foo} and \metavar{bar} are implicit groups. \\

\code{\metavar{foo}\,\metavar{bar}} &
Sequence; \metavar{foo} followed by \metavar{bar}. \\

\code{\metavar{foo}\metaor\metavar{bar}} &
Choice; \metavar{foo} or \metavar{bar}. \\

\metavar{foo}\kleene* &
\metavar{foo} can occur zero or more times. \\

\metavar{foo}\kleene+ &
\metavar{foo} can occur one or more times. \\

\metavar{foo}\kleene? &
\metavar{foo} is optional. \\

\group{\,} &
Explicit group. \\

\code{\defkey{}\metavar{key} = \metavar{value}} &
User option, \metavar{key} and \metavar{value} are implicit
groups. \\

\code{\defkey{}\metavar{key}} &
User option without a value. \\

\default{42} &
Default value is 42. \\

\carriagereturn{} &
Line continuation mark. \\

\metavar{empty} &
Nothing. \\

\metavar{newline} &
Newline character,
\code{\textasciicircum\textasciicircum M} in \TeX{}. \\

\metavar{dimension} &
A legitimate \TeX{} dimension. \\

\metavar{number} &
$(-\infty,\,\infty)\cap\mathbb{R}$. \\

\metavar{positive number} &
$(0,\,\infty)\cap\mathbb{R}$. \\

\metavar{non-negative number} &
$[0,\,\infty)\cap\mathbb{R}$. \\

\metavar{integer} &
$(-\infty,\,\infty)\cap\mathbb{Z}$. \\

\metavar{positive integer} &
$(0,\,\infty)\cap\mathbb{Z}$. \\

\metavar{non-negative integer} &
$[0,\,\infty)\cap\mathbb{Z}$. \\
\end{tabulary}
